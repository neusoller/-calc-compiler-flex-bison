h := d or f            // Boolean: true, perquè d=true

// Expressions aritmètiques
i := a + b             // Real: suma d'un enter i un real
j := (a * 2) - b       // Real: multiplicació d'enter i resta amb real
k := c + " món"        // Cadena: concatenació de cadena amb literal

// Casos especials
l := b / 0             // Error: divisió per zero
m := k + 123           // Cadena: concatenació de cadena i enter
